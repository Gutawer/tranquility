class TQ_Pistol : Weapon {
	bool SlideLocked;
	bool SlideBack;
	bool RoundInChamber;
	bool SafetyOn;
	bool MagIn;

	Default	{
		Weapon.SlotNumber 2;
	}

	override void PostBeginPlay() {
		RoundInChamber = 1;
		RoundsInMag = 7;
	}

	States {
		Spawn:
			PISG P -1;
			Loop;
		Select:
			TNT1 A 1 A_Raise;
			Loop;
		Deselect:
			TNT1 A 1 A_Print("Deselect");
			TNT1 A 1 A_Lower;
			Loop;
		Ready:
			TNT1 A 1 InputChecker();
			Loop;
		Fire:
			TNT1 A 5 {
				if(invoker.RoundInChamber==true) {
					A_Print("Fire");
					invoker.RoundInChamber=false;
				}
			}
			TNT1 A 0 A_Print("");
			Goto Ready;
	}

	//Credit to SanyaWaffles via Discord
	action bool CheckButtonPress(int btn) {
		return ((GetPlayerInput(INPUT_BUTTONS) & btn) && !(GetPlayerInput(INPUT_OLDBUTTONS) & btn));
	}

	action state InputChecker() {
		A_WeaponReady(WRF_NOFIRE|WRF_NOPRIMARY|WRF_NOSECONDARY);
		if(CheckButtonPress(BT_ATTACK)) {
			return ResolveState("Fire");
		}
		return ResolveState(null);
	}

	//ChamberFromMag

}
